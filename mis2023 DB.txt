## Member
CREATE TABLE TBL_MEMBER (
    USERID VARCHAR2(50) NOT NULL,
    USERPW VARCHAR2(50) NOT NULL,
    USERNAME VARCHAR2(50) NOT NULL,
    EMAIL VARCHAR2(100),
    REGDATE DATE DEFAULT SYSDATE,
    UPDATEDATE DATE DEFAULT SYSDATE,
    PRIMARY KEY (USERID)
);

select * from TBL_MEMBER;

SELECT * FROM TBL_MEMBER
				WHERE USERID = ''
                
SELECT * FROM TBL_MEMBER
				WHERE USERID = '' AND USERPW = ''

## Board 
CREATE TABLE TBL_BOARD (
	BNO NUMBER NOT NULL, 
	TITLE VARCHAR2(200) NOT NULL, 
	CONTENT VARCHAR2(1000) NOT NULL, 
	WRITER VARCHAR2(50) NOT NULL, 
	REGDATE DATE DEFAULT SYSDATE, 
	VIEWCNT NUMBER DEFAULT 0, 
	PRIMARY KEY (BNO)
 );
 
CREATE SEQUENCE SEQ_BOARD INCREMENT BY 1 START WITH 1;

select * from TBL_BOARD;

--페이징 테스트를 위해 데이터 INSERT
INSERT INTO TBL_BOARD( bno, title, content, writer, regdate, viewcnt)
SELECT SEQ_BOARD.nextval, title, content, writer, regdate, viewcnt FROM TBL_BOARD;

SELECT * FROM TBL_BOARD WHERE bno > 0 and (rownum >= 1 AND rownum <= 10);
